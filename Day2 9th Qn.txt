Q9.Write a C Program to print two numbers with a tab space between them.

#include <stdio.h>

int main() {
    int num1, num2;

    // Input two numbers
    printf("Enter two numbers: ");
    scanf("%d %d", &num1, &num2);

    // Print with tab space
    printf("%d\t%d\n", num1, num2);

    return 0;
}
